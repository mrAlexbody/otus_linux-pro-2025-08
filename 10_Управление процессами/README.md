### Своя реализация lsof

```shell
#!/bin/bash
printf "%-15s %-8s %-10s %s\n" "COMMAND" "PID" "USER" "FILE"
for pid in /proc/[0-9]*; do
    pid=${pid##*/}
    cmd=$(</proc/$pid/comm 2>/dev/null) || continue
    user=$(stat -c "%U" /proc/$pid 2>/dev/null) || user="unknown"
    for fd in /proc/$pid/fd/* ; do
        file=$(readlink -f $fd 2>/dev/null)
        [[ $file && $file != "/dev/null" ]] && echo "$cmd $pid $user $file"
    done
done | column -t
```
